#####################################
To upload files after repo creation
##################################### 

:: Make sure we are in the project folder
cd /d D:\softdata_consult\Projects\R_spatial_analysis

:: Initialize Git if not already done
git init

:: Stage all files
git add .

:: Commit (only if no commit yet, safe to run again)
git commit -m "Initial commit"

:: Rename branch to main (fixes 'master' vs 'main' issue)
git branch -M main

:: Set the GitHub repo as the remote

git remote add origin https://github.com/softdataconsult/R_spatial_analysis.git

:: Push everything to GitHub

git push -u origin main
